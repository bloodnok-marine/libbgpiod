178 static void
179 print_gpioline(bgpio_chip_t *chip, int line_no)
180 {
181  struct gpio_v2_line_info *info;
182  char line[200] = "";
183  uint64_t attr_flags;
184  uint64_t output_values;
185  char o_val[32];
186  uint32_t debounce;
187 
188  line[0] = '\0';
189  info = bgpio_get_lineinfo(chip, line_no);
190  printf("%3d: ", line_no);
191  append_at(line, 0, info->name);
192  if (BGPIO_MASKED_BITS(info->flags, GPIO_V2_LINE_FLAG_USED)) {
193  append_at(line, 20, "\"");
194  append_at(line, 21, info->consumer);
195  append_at(line, 20, "\"");
196  }
197  else {
198  append_at(line, 20, "unused");
199  }
200  append_at(line, 36, "");
201  attr_flags = bgpio_attr_flags(info);
202  append_flags(line, info->flags, attr_flags);
203  if (bgpio_attr_output(info, &output_values)) {
204  sprintf(o_val, " [0x%" PRIx64 "]", output_values);
205  (void) strcat(line, o_val);
206  }
207  if (bgpio_attr_debounce(info, &debounce)) {
208  sprintf(o_val, " (%dÎ¼sec)", debounce);
209  (void) strcat(line, o_val);
210  }
211  printf("%s\n", line);
212  free((void *) info);
213 }

257 int
258 main(int argc, char *argv[])
259 {
263  static struct option options[] = {
264  {"help", no_argument, 0, 0},
265  {"version", no_argument, 0, 0},
266  {0, 0, 0, 0}};
267 
268  int c;
269  int idx = 0;
270  svector *paths;
271 
272  c = getopt_long(argc, argv, "hv", options, &idx);
273  if (c != -1) {
274  switch (c) {
275  case 0:
276  if (streq("version", options[idx].name)) {
277  version(THIS_EXECUTABLE);
278  }
279  if (streq("help", options[idx].name)) {
280  usage(0);
281  }
282  fprintf(stderr, "%s: unhandled option: %s\n\n",
283  THIS_EXECUTABLE, options[idx].name);
284  usage(EINVAL);
285  case 'v':
286  version("THIS_EXECUTABLE");
287  case 'h':
288  usage(0);
289  default:
290  usage(EINVAL);
291  }
292  }
293 
294  paths = get_chip_paths();
295 
296  if (argc > 1) {
297  char *device;
298  device = path_for_arg(paths, argv[1]);
299  if (device) {
300  device = newstrcpy(device);
301  }
302  else {
303  device = newstrcpy(argv[1]);
304  fprintf(stderr,
305  "%s may not be a gpio device. Trying anyway...\n",
306  argv[1]);
307  }
308  free_chip_paths(paths);
309  /* Replace the original paths ::svector with a version with
310  * only the single device. */
311  paths = create_svector(argc);
312  paths = svector_add_elem(paths, device);
313  }
314 
315  for (idx = 0; idx < paths->elems; idx++) {
316  /* We pass in argc and argv so that specific lines can be
317  * queried if the user has specified them. */
318  bgpio_chip_t *chip = bgpio_open_chip(paths->str[idx]);
319  int line;
320  if (chip) {
321  printf("%s - %d lines\n",
322  chip->info.name, chip->info.lines);
323 
324  if (argc > 2) {
325  int i;
326  for (i = 2; i < argc; i++) {
327  line = get_gpio_line(argv[i], chip->info.lines);
328  print_gpioline(chip, line);
329  }
330  }
331  else {
332  for (line = 0; line < chip->info.lines; line++) {
333  print_gpioline(chip, line);
334  }
335  }
336  bgpio_close_chip(chip);
337  }
338  else {
339  fprintf(stderr, "%s: unable to open %s (%s)\n",
340  THIS_EXECUTABLE, paths->str[idx], strerror(errno));
341  exit(errno);
342  }
343  }
344  free_chip_paths(paths);
345 }
