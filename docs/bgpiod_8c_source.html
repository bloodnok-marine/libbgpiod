67 static void
68 bgpio_clear_line_flags(bgpio_request_t *req, int line)
69 {
70  int line_idx = bgpio_idx_for_line(req, line);
71  bool reclaim_some_attrs = false;
72  for (int i = 0; i < req->req.config.num_attrs; i++) {
73  if (BGPIO_BITVALUE(req->req.config.attrs[i].mask, line_idx)) {
74  BGPIO_CLEARBIT(req->req.config.attrs[i].mask, line_idx);
75  if (req->req.config.attrs[i].mask == 0) {
76  /* We have removed the last user of the current
77  * line_config_attribute. Mark it to be reclaimed
78  * later. */
79  req->req.config.attrs[i].attr.id = 0;
80  reclaim_some_attrs = true;
81  }
82  }
83  }
84  if (reclaim_some_attrs) {
85  int target = 0;
86  int source = 0;
87  while (true) {
88  if (req->req.config.attrs[source].attr.id == 0) {
89  source++;
90  }
91  else {
92  if (source != target) {
93  req->req.config.attrs[target] =
94  req->req.config.attrs[source];
95  }
96  source++;
97  target++;
98  }
99  if (source >= req->req.config.num_attrs) {
100  req->req.config.num_attrs = target;
101  return;
102  }
103  }
104  }
105 }

123 static int
124 bgpio_set_line_flags(bgpio_request_t *req, int line, uint64_t flags)
125 {
126  int i;
127  int line_idx = bgpio_idx_for_line(req, line);
128  bgpio_clear_line_flags(req, line);
129 
130  for (i = 0; i < req->req.config.num_attrs; i++) {
131  /* If other lines are already using the same set of
132  * flag attributes, we will just add this line to the
133  * record. */
134  if ((req->req.config.attrs[i].attr.id == GPIO_V2_LINE_ATTR_ID_FLAGS) &&
135  (req->req.config.attrs[i].attr.flags == flags)) {
136  BGPIO_SETBIT(req->req.config.attrs[i].mask, line_idx);
137  return i;
138  }
139  }
140  if (req->req.config.num_attrs < GPIO_V2_LINE_NUM_ATTRS_MAX) {
141  i = req->req.config.num_attrs;
142  req->req.config.num_attrs++;
143  req->req.config.attrs[i].attr.id = GPIO_V2_LINE_ATTR_ID_FLAGS;
144  req->req.config.attrs[i].attr.flags = flags;
145  req->req.config.attrs[i].mask = BGPIO_BITMASK(line_idx);
146  return i;
147  }
148  return EINVAL;
149 }

167 static bool
168 bgpio_update_initial_value(bgpio_request_t *req, int line, int output_value)
169 {
170  int attr_idx = -1;
171  int line_idx = bgpio_idx_for_line(req, line);
172  for (int i = 0; i < req->req.config.num_attrs; i++) {
173  if (req->req.config.attrs[i].attr.id ==
174  GPIO_V2_LINE_ATTR_ID_OUTPUT_VALUES) {
175  attr_idx = 0;
176  break;
177  }
178  }
179  if (attr_idx < 0) {
180  if (req->req.config.num_attrs >= GPIO_V2_LINE_NUM_ATTRS_MAX) {
181  return false;
182  }
183  attr_idx = req->req.config.num_attrs;
184  req->req.config.num_attrs++;
185  req->req.config.attrs[attr_idx].attr.values = 0;
186  }
187  req->req.config.attrs[attr_idx].attr.id =
188  GPIO_V2_LINE_ATTR_ID_OUTPUT_VALUES;
189  BGPIO_SETBIT(req->req.config.attrs[attr_idx].mask, line_idx);
190  if (output_value) {
191  BGPIO_SETBIT(req->req.config.attrs[attr_idx].attr.values, line_idx);
192  }
193  return true;

259 char *
260 bgpio_configure_line(
261  bgpio_request_t *req, int line, uint64_t flags, ...)
262 {
263  assert(req);
264  int res;
265  int idx = bgpio_idx_for_line(req, line);
266  int output_value;
267  struct gpio_v2_line_info line_info;
268  char *result;
269  memset((void *) &line_info, 0, sizeof(line_info));
270  if (idx == -1) {
271  idx = req->req.num_lines;
272  req->req.offsets[idx] = line;
273  req->req.num_lines++;
274  if (req->req.num_lines >= GPIO_V2_LINES_MAX) {
275  fprintf(stderr, "bgpio_configure_line: max lines (%d) exceeded.\n",
276  GPIO_V2_LINES_MAX);
277  errno = EINVAL;
278  return NULL;
279  }
280 
281  line_info.offset = line;
282  res = ioctl(req->device_fd, GPIO_V2_GET_LINEINFO_IOCTL,
283  (void *) &line_info);
284 
285  if (res < 0) {
286  return NULL;
287  }
288  if (line_info.consumer[0]) {
289  /* A consumer has been registered for this line. */
290  fprintf(stderr,
291  "bgpio_configure_line: line %d is registered to \"%s\"\n",
292  line, line_info.consumer);
293  errno = EBUSY;
294  return NULL;
295  }
296  }
297 
298  BGPIO_SETBIT(req->line_values.mask, idx);
299  if (flags) {
300  res = bgpio_set_line_flags(req, line, flags);
301  if (res < 0) {
302  fprintf(stderr, "bgpio_configure_line: flags failure.\n");
303  errno = EINVAL;
304  return NULL;
305  }
306  }
307 
308  if (BGPIO_MASKED_BITS(flags, GPIO_V2_LINE_FLAG_OUTPUT)) {
309  va_list ap;
310  va_start(ap, flags);
311  output_value = va_arg(ap, int);
312  va_end(ap);
313  if ((output_value != 0) && (output_value != 1)) {
314  fprintf(stderr, "INVALID OUTPUT VALUE\n");
315  }
316  if (!bgpio_update_initial_value(req, line, output_value)) {
317  errno = EINVAL;
318  return NULL;
319  }
320  }
321 
322  result = malloc(strlen(line_info.name) + 1);
323  strcpy(result, line_info.name);
324  return result;
325 }

775 int
776 bgpio_await_event(bgpio_request_t *req,
777  int *timeout_msecs)
778 {
779  int res;
780  if (timeout_msecs) {
781  struct pollfd poll_fd = {req->req.fd, POLLIN, 0};
782  res = poll(&poll_fd, 1, *timeout_msecs);
783 
784  if (res == 0) {
785  /* We timed-out. Let the caller know. */
786  return ETIMEDOUT;
787  }
788  if (res < 0) {
789  return errno? errno: EINVAL;
790  }
791  if (!poll_fd.revents & POLLIN) {
792  /* No input data available. This is bad but we cannot
793  * easily provide more information here. */
794  return EINVAL;
795  }
796  }
797 
798  res = read(req->req.fd, &(req->event),
799  sizeof(struct gpio_v2_line_event));
800  if (res == -1) {
801  return errno;
802  }
803  if (res != sizeof(struct gpio_v2_line_event)) {
804  return EINVAL;
805  }
806  return 0;
807 }
