44 static void
45 usage(int exitcode)
46 {
47  printf("\n"
48  "Usage: " THIS_EXECUTABLE " [OPTIONS] <chip-id> <line-spec>...\n\n"
49  "Get input from GPIO lines.\n\n"
50  "Options:\n -b, --bias=[as-is|disable|pull-down|pull-up]\n"
51  " set the line bias (default=as-is)\n"
52  " -d, --delta: report only when state changes\n"
53  " -h, --help: display this help message.\n"
54  " -l, --active-low, --low: "
55  "make the line active-low (default).\n"
56  " -n, --name=our_name: who has reserved our gpio lines \n"
57  " -p, --period=usecs: period for loop (default=2000000)\n"
58  " -q, --quiet: execute quietly\n"
59  " -r, --repeat=count: how many times to fetch (default=1)\n"
60  " -v, --version: display the version.\n"
61  " -x, --exec=path: command to execute on change\n\n");
62  if (!exitcode) {
63  printf(
64  "Gpiochip-ids may be a full path to the gpiochip device, or an\n"
65  "abbeviated suffix (eg \"chip0\") of a valid path.\n\n"
66  "Line-specs are of the form N[\"[\"line-flag[,line-flag]\"]\"]\n"
67  "where line-flag may be a bias value, active-high, high or \n"
68  "active-low, eg 42[pull-down] 43[pull-up,active-high].\n\n"
69  "Specifying a repeat value of zero means repeat forever.\n\n"
70  "The command executed by the exec option will be passed the\n"
71  "gpio device path, the gpio line number and the gpio line value\n"
72  "as parameters.\n\n"
73  "The result of the command will be the value of the last\n"
74  "successful gpio fetch, or an errorcode if an error occurred.\n");
75  }
76  exit(exitcode);
77 }

207 static int
208 perform_fetches(bgpio_request_t *request,
209  bool quiet, bool report_delta,
210  char *exec, char *names[])
211 {
212  static uint64_t previous;
213  static bool first_time = true;
214  int result;
215  int i;
216  int val = 0;
217  int line;
218  bool report;
219  result = bgpio_fetch(request);
220  if (result < 0) {
221  fprintf(stderr, "%s: bgpio_failed (%s)\n",
222  THIS_EXECUTABLE, strerror(errno));
223  exit(errno);
224  }
225 
226  if (!(first_time && report_delta)) {
227  /* If we are reporting delta only, and this is the first time
228  * through this function, we do not need to report anything,
229  * so we do not execute this block of code. */
230  for (i = 0; i < request->req.num_lines; i++) {
231  val = bgpio_fetched_by_idx(request, i, &line);
232  if (report_delta) {
233  report = BGPIO_BITVALUE(previous, i) != val;
234  }
235  else {
236  report = true;
237  }
238  if (report) {
239  if (!quiet) {
240  printf("Line %d (%s) = %d\n", line, names[i], val);
241  }
242  if (exec) {
243  char *command_str = malloc(strlen(exec) + 30);
244  int err;
245  sprintf(command_str, "%s %s %d %d", exec,
246  request->chardev_path, line, val);
247  err = system(command_str);
248  if (err) {
249  fprintf(stderr, "%s: \"%s\" failed: %d\n\n",
250  THIS_EXECUTABLE, command_str, err);
251  }
252  }
253  }
254  }
255  }
256  first_time = false;
257  previous = request->line_values.bits;
258  return val;
259 }
