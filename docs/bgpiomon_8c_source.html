70  " -r, --repeat=count how many edges to detect (default=1)\n"
71  " -t, --timeout=millisecs Specify an inactivity timeout period.\n"
72  " -v, --version: display the version.\n"
73  " -x, --exec=path: command to execute on detection\n\n");

75  printf(

254 static int
255 process_edge(bgpio_request_t *request, bool quiet,
256  char *exec, int *timeout)
257 {
258  struct gpio_v2_line_event *p_event = &(request->event);
259  int result;
260 
261  if ((result = bgpio_await_event(request, timeout))) {
262  // TODO: Put in proper error message
263  if (result == ETIMEDOUT) {
264  return 0;
265  }
266  fprintf(stdout, "%s: Await event error: %d\n",
267  THIS_EXECUTABLE, result);
268  exit(result);
269  }
270 
271  if (!quiet) {
272  fprintf(stdout, "GPIO EVENT at %" PRIu64 " on line %d (%d|%d) ",
273  (uint64_t)p_event->timestamp_ns, p_event->offset,
274  p_event->line_seqno, p_event->seqno);
275  }
276  switch (p_event->id) {
277  case GPIO_V2_LINE_EVENT_RISING_EDGE:
278  if (!quiet) {
279  fprintf(stdout, "rising edge\n");
280  }
281  result = 1;
282  break;
283  case GPIO_V2_LINE_EVENT_FALLING_EDGE:
284  if (!quiet) {
285  fprintf(stdout, "falling edge\n");
286  }
287  result = 0;
288  break;
289  default:
290  fprintf(stderr, "%s: unknown event type: %d\n",
291  THIS_EXECUTABLE, p_event->id);
292  return EINVAL;
293  }
294 
295  if (exec) {
296  char *command_str = malloc(strlen(exec) + 80);
297  int err;
298  sprintf(command_str, "%s %s %d %d %lld %d %d",
299  exec, request->chardev_path,
300  p_event->offset, result,
301  p_event->timestamp_ns,
302  p_event->line_seqno, p_event->seqno);
303  err = system(command_str);
304  if (err) {
305  fprintf(stderr, "%s: \"%s\" failed: %d\n\n",
306  THIS_EXECUTABLE, command_str, err);
307  }
308  }
309  return result;
310 }

316 int
317 main(int argc, char *argv[])
318 {
319  int active_low = false;
320  char *consumer_name = THIS_EXECUTABLE;
321  char *exec = NULL;
322  int quiet = false;
323  int repeat = 1;
324  int timeout = -1;
325  uint64_t default_bias = 0;
326  uint64_t default_edge = GPIO_V2_LINE_FLAG_EDGE_RISING;
327  unsigned long debounce_period = 0;
328 
329  struct option options[] = {
330  {"active-low", no_argument, &active_low, true},
331  {"bias", required_argument, NULL, 0},
332  {"debounce", required_argument, NULL, 0},
333  {"edge", required_argument, NULL, 0},
334  {"exec", required_argument, NULL, 0},
335  {"help", no_argument, 0, 0},
336  {"low", no_argument, &active_low, true},
337  {"name", required_argument, NULL, 0},
338  {"quiet", no_argument, &quiet, true},
339  {"repeat", required_argument, NULL, 0},
340  {"timeout", required_argument, NULL, 0},
341  {"version", no_argument, 0, 0},
342  {0, 0, 0, 0}
343  };
344 
345  int c;
346  int idx = 0;
347  bgpio_request_t *request;
348  uint64_t line_flags;
349  uint64_t all_lines_mask = 0;
350  char *line_name;
351  int line;
352  int result = 0;
353  int err;
354 
355  while ((c = getopt_long(argc, argv, "b:d:e:hln:qr:t:vx:",
356  options, &idx)) != -1)
357  {
358  switch (c) {
359  case 0:
360  if (streq("version", options[idx].name)) {
361  version(THIS_EXECUTABLE);
362  }
363  if (streq("help", options[idx].name)) {
364  usage(0);
365  }
366  if (streq("active-low", options[idx].name) ||
367  streq("low", options[idx].name) ||
368  streq("quiet", options[idx].name)) {
369  }
370  else if (streq("bias", options[idx].name)) {
371  default_bias = get_bias(optarg);
372  }
373  else if (streq("debounce", options[idx].name)) {
374  debounce_period = get_debounce(optarg);
375 #ifndef DEBOUNCE_DISABLED
376  fprintf(stderr,
377  "%s: Warning debounce is not yet enabled - ignored.\n",
378  THIS_EXECUTABLE);
379 #endif
380  }
381  else if (streq("edge", options[idx].name)) {
382  default_edge = get_edge(optarg);
383  }
384  else if (streq("exec", options[idx].name)) {
385  exec = optarg;
386  }
387  else if (streq("name", options[idx].name)) {
388  consumer_name = optarg;
389  }
390  else if (streq("repeat", options[idx].name)) {
391  repeat = get_repeat(optarg);
392  }
393  else if (streq("timeout", options[idx].name)) {
394  timeout = get_timeout(optarg);
395  }
396  else {
397  fprintf(stderr, "%s: unhandled option: %s\n\n",
398  THIS_EXECUTABLE, options[idx].name);
399  usage(EINVAL);
400  }
401  break;
402  case 'b':
403  default_bias = get_bias(optarg);
404  continue;
405  case 'd':
406  debounce_period = get_debounce(optarg);
407 #ifdef DEBOUNCE_DISABLED
408  fprintf(stderr,
409  "%s: Warning debounce is not yet enabled - ignored.\n",
410  THIS_EXECUTABLE);
411  continue;
412 #endif
413  case 'e':
414  default_edge = get_edge(optarg);
415  continue;
416  case 'h':
417  usage(0);
418  case 'l':
419  active_low = true;
420  continue;
421  case 'n':
422  consumer_name = optarg;
423  continue;
424  case 'q':
425  quiet = true;
426  continue;
427  case 'r':
428  repeat = get_repeat(optarg);
429  continue;
430  case 't':
431  timeout = get_timeout(optarg);
432  continue;
433  case 'v':
434  version("THIS_EXECUTABLE");
435  case 'x':
436  exec = optarg;
437  continue;
438  default:
439  usage(EINVAL);
440  }
441  }
442 
443  if (optind >= argc) {
444  fprintf(stderr, "%s: No gpio chip id provided.\n", THIS_EXECUTABLE);
445  usage(EINVAL);
446  }
447 
448  request = get_gpio_request(argv[optind], consumer_name, 0);
449 
450  /* Now handle each line argument in turn. */
451  if (((argc - 1) - optind) > GPIO_V2_LINES_MAX) {
452  fprintf(stderr, "%s: maximum gpio lines (%d) exceeded.\n",
453  THIS_EXECUTABLE, GPIO_V2_LINES_MAX);
454  exit(EINVAL);
455  }
456 
457  for (idx = optind + 1; idx < argc; idx++) {
458  line_flags = GPIO_V2_LINE_FLAG_INPUT |
459  default_bias |
460  default_edge |
461  (active_low? GPIO_V2_LINE_FLAG_ACTIVE_LOW: 0);
462 
463  if (!read_line_arg(argv[idx], &line, &line_flags,
464  LINE_FLAG_BIAS_MASK |
465  LINE_FLAG_EDGE_MASK |
466  LINE_FLAG_ACTIVE_LOW_MASK))
467  {
468  fprintf(stderr, "expecting numeric gpio line with "
469  "optional bias: \"%s\"\n", argv[idx]);
470  usage(EINVAL);
471  }
472 
473  line_name = bgpio_configure_line(request, line, line_flags);
474 
475  if (!line_name) {
476  fprintf(stderr, "%s: unable to get line (%d) for chip\n",
477  THIS_EXECUTABLE, line);
478  exit(EINVAL);
479  }
480 
481  /* Add bits for each line into all_lines_mask */
482  all_lines_mask |= 1ull << (idx - (optind + 1));
483  }
484 
485 #ifdef DEBOUNCE_DISABLED
486  // Keep the compiler from reporting an unused variable.
487  debounce_period++;
488 #else
489  if (debounce_period) {
490  struct gpio_v2_line_config *config = &(request->req.config);
491  int attr = config->num_attrs;
492  /* For now we only allow a global debounce period rather than
493  * different ones for different lines.
494  */
495  config->num_attrs++;
496  config->attrs[attr].attr.id = GPIO_V2_LINE_ATTR_ID_DEBOUNCE;
497  config->attrs[attr].attr.debounce_period_us = debounce_period;
498  config->attrs[attr].mask = all_lines_mask;
499  }
500 #endif
501  if (request->req.num_lines) {
502  result = bgpio_complete_request(request);
503 
504  if (result) {
505  fprintf(stderr, "%s: error completing bgpio_request: %s\n",
506  THIS_EXECUTABLE, strerror(errno));
507  exit(errno);
508  }
509 
510  idx = repeat;
511  while (true) {
512  result = process_edge(request, quiet, exec,
513  (timeout == -1? NULL: &timeout));
514  if ((result == 0) || (result == 1)) {
515 
516  if (repeat) {
517  /* If repeat is zero we want an infinite number of
518  * repeats, so we don't do the decrement and
519  * conditional break. */
520  idx--;
521  if (idx < 1) {
522  break;
523  }
524  }
525  }
526  }
527  }
528  err = bgpio_close_request(request);
529  if (err) {
530  fprintf(stderr, "%s: error closing bgpio_request: %s\n",
531  THIS_EXECUTABLE, strerror(errno));
532  exit(errno);
533  }
534  exit(result);
535 }
