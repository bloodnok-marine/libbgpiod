157 static int
158 watch_lines(bgpio_chip_t *chip, int repeat, int *timeout,
159  char *exec, bool quiet)
160 {
161  struct gpio_v2_line_info_changed *event;
162  char *event_str;
163 
164  while (true) {
165  event = bgpio_await_watched_lines(chip, timeout);
166  if (event) {
167  switch (event->event_type) {
168  case GPIO_V2_LINE_CHANGED_REQUESTED:
169  event_str = "requested";
170  break;
171  case GPIO_V2_LINE_CHANGED_RELEASED:
172  event_str = "released";
173  break;
174  case GPIO_V2_LINE_CHANGED_CONFIG:
175  event_str = "config changed";
176  break;
177  default:
178  fprintf(stderr,
179  "%s: Invalid event type (%d) received from kernel\n",
180  THIS_EXECUTABLE, event->event_type);
181  bgpio_close_chip(chip);
182  return EINVAL;
183  }
184 
185  if (!quiet) {
186  printf("line %u: %s at %" PRIu64 "\n",
187  event->info.offset, event_str,
188  (uint64_t)event->timestamp_ns);
189  }
190  if (exec) {
191  char *command_str = malloc(strlen(exec) + 120);
192  int err;
193  sprintf(command_str, "%s %s %d \"%s\" %" PRIu64 "\n",
194  exec, chip->path,
195  event->info.offset, event_str,
196  (uint64_t)event->timestamp_ns);
197  err = system(command_str);
198  if (err) {
199  if (!quiet) {
200  fprintf(stderr,
201  "%s: command \"%s\" failed (%d).\n",
202  THIS_EXECUTABLE, command_str, err);
203  return err;
204  }
205  }
206  }
207  }
208  else {
209  if (errno) {
210  fprintf(stderr, "%s: Watch failed: %s\n",
211  THIS_EXECUTABLE, strerror(errno));
212  return errno;
213  }
214  }
215 
216  if (repeat) {
217  repeat--;
218  if (!repeat) {
219  return 0;
220  }
221  }
222  }
223 }

227 {

242  struct option options[] = {

244  {"help", no_argument, NULL, 0},
245  {"quiet", no_argument, &quiet, true},

251  while ((c = getopt_long(argc, argv, "hqr:t:vx:",
252  options, &idx)) != -1)
253  {
254  switch (c) {
255  case 0:
256  if (streq("version", options[idx].name)) {
257  version(THIS_EXECUTABLE);
258  }
259  if (streq("help", options[idx].name)) {
260  usage(0);
261  }
262  if (streq("quiet", options[idx].name)) {
263  }
264  else if (streq("exec", options[idx].name)) {
265  exec = optarg;
266  }
267  else if (streq("repeat", options[idx].name)) {
268  repeat = get_repeat(optarg);
269  }
270  else if (streq("timeout", options[idx].name)) {
271  timeout = get_timeout(optarg);
272  }
273  else {
274  fprintf(stderr, "%s: unhandled option: %s\n\n",
275  THIS_EXECUTABLE, options[idx].name);
276  usage(EINVAL);
277  }
278  break;
279  case 'h':
280  usage(0);
281  case 'q':
282  quiet = true;
283  continue;
284  case 'r':
285  repeat = get_repeat(optarg);
286  continue;
287  case 't':
288  timeout = get_timeout(optarg);
289  continue;
290  case 'v':
291  version("THIS_EXECUTABLE");
292  case 'x':
293  exec = optarg;
294  continue;
295  default:
296  usage(EINVAL);
297  }
298  }
299 
300  if (optind >= argc) {
301  fprintf(stderr, "%s: No gpio chip id provided.\n", THIS_EXECUTABLE);
302  usage(EINVAL);
303  }
304 
305  chip = get_gpio_chip(argv[optind]);
306  if (chip) {
307  for (idx = optind + 1; idx < argc; idx++) {
308  if (read_int(argv[idx], &line)) {
309  err = bgpio_watch_line(chip, line);
310  if (err) {
311  fprintf(stderr, "%s: unable to watch line %d: %s.\n",
312  THIS_EXECUTABLE, line, strerror(err));
313  bgpio_close_chip(chip);
314  return err;
315  }
316  lines++;
317  }
318  else {
319  fprintf(stderr, "%s: invalid line value \"%s\".\n",
320  THIS_EXECUTABLE, argv[idx]);
321 
322  bgpio_close_chip(chip);
323  return EINVAL;
324  }
325  }
326  if (lines) {
327  err = watch_lines(chip, repeat,
328  (timeout == -1)? NULL: &timeout,
329  exec, quiet);
330  }
331  bgpio_close_chip(chip);
332  }
333 
334  return errno;
335 }
