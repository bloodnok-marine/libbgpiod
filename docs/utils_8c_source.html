107 char *
108 path_for_arg(svector *paths, char *arg)
109 {
110  finder_fn_t finder;
111  int match;
112  if (arg) {
113  if (!paths) {
114  /* Note that we do not free the svector returned from
115  * here. This is a conscious choice, doing so would
116  * require much more code and buy us very little.
117  */
118  paths = get_chip_paths();
119  }
120  if (arg[0] == '/') {
121  /* arg looks like a full path, so use strcmp as the
122  * comparision function */
123  finder = strcmp;
124  }
125  else {
126  finder = endcmp;
127  }
128  match = svector_find(paths, arg, finder);
129  if (match < 0) {
130  return NULL;
131  }
132  return paths->str[match];
133  }
134  return NULL;
135 }

379 bool
380 parse_lineflags(char *arg, uint64_t *flags, uint64_t allowed_flags)
381 {
382  char *comma;
383  char *bracket = NULL;
384  bool bias_allowed = (allowed_flags & LINE_FLAG_BIAS_MASK) != 0;
385  bool odrive_allowed = (allowed_flags & LINE_FLAG_OUTPUT_DRIVER_MASK) != 0;
386  bool edge_allowed = (allowed_flags & LINE_FLAG_EDGE_MASK) != 0;
387  bool active_allowed = (allowed_flags & LINE_FLAG_ACTIVE_LOW_MASK) != 0;
388  do {
389  comma = strchr(arg, ',');
390  if (comma) {
391  *comma = '\0'; /* Temporaily truncate string */
392  }
393  else {
394  bracket = strchr(arg, ']');
395  *bracket = '\0'; /* Temporarily truncate string */
396  }
397 
398  if (!((bias_allowed && strbias(arg, flags)) ||
399  (odrive_allowed && stroutputdrive(arg, flags)) ||
400  (edge_allowed && stredge(arg, flags)) ||
401  (active_allowed && stractive(arg, flags))))
402  {
403  /* Nothing allowed and valid was read. Oh dear. */
404  if (bracket) {
405  *bracket = ']'; /* Restore arg to its orignal state. */
406  }
407  return false;
408  }
409 
410  if (comma) {
411  *comma = ','; /* Restore arg to its orignal state. */
412  arg = comma + 1; /* Set arg to the start of next parameter */
413  }
414  } while (comma);
415 
416  if (bracket) {
417  *bracket = ']'; /* Restore arg to its orignal state. */
418 
419  }
420 
421  return true;
422 }
