53  "Options:\n -b, --bias=[as-is|disable|pull-down|pull-up]\n"

62  printf(
63  "Chip-ids may be a full path to the gpiochip device, or an\n"
64  "abbeviated suffix (eg \"chip0\") of a valid path.\n\n"
65  "Line-specs are of the form N[\"[\"line-flag[,line-flag...]\"]\"]=B\n"
66  "where line-flag may be a bias value, output-drive-value, \n"
67  "active-high, high or active-low, N is the gpio line number and B\n"
68  "is the binary digit 1 or 0, eg \"84[open-drain,high]=1\"\n");
69  }
70  exit(exitcode);

180 static bool
181 read_line_spec(char *arg, int *line, int *value,
182  uint64_t *line_flags, uint64_t allowed)
183 {
184  char flags_str[2];
185  int fields;
186 
187  fields = sscanf(arg, "%d=%d", line, value);
188  if (fields == 2) {
189  return true;
190  }
191  fields = sscanf(arg, "%d[%1s", line, flags_str);
192  if (fields == 2) {
193  char *bracket = strchr(arg, '[');
194  arg = bracket + 1;
195  if (!parse_lineflags(arg, line_flags, allowed)) {
196  return false;
197  }
198  bracket = strchr(arg, ']');
199  fields = sscanf(bracket, "]=%d", value);
200  if (fields == 1) {
201  if ((*value == 0) || (*value == 1)) {
202  return true;
203  }
204  }
205  }
206  return false;
207 }

210 int main(int argc, char *argv[])
211 {
212  int active_low = false;
213  char *consumer_name = THIS_EXECUTABLE;
214  int seconds = 0;
215  int wait = false;
216  uint64_t base_flags = GPIO_V2_LINE_FLAG_OUTPUT;
217  uint64_t line_flags = 0;
218  int line_value;
219 
223  struct option options[] = {
224  {"bias", required_argument, NULL, 0},
225  {"active-low", no_argument, &active_low, true},
226  {"help", no_argument, NULL, 0},
227  {"low", no_argument, &active_low, true},
228  {"name", required_argument, NULL, 0},
229  {"output-drive", required_argument, NULL, 0},
230  {"version", no_argument, NULL, 0},
231  {"wait", required_argument, &wait, 0},
232  {NULL, 0, NULL, 0}};
233 
234  int c;
235  int idx = 0;
236  bgpio_request_t *request;
237  int line;
238  char *line_name;
239  int err;
240  uint64_t drive;
241 
242  while ((c = getopt_long(argc, argv, "b:hln:o:vw:",
243  options, &idx)) != -1)
244  {
245  switch (c) {
246  case 0:
247  if (streq("version", options[idx].name)) {
248  version(THIS_EXECUTABLE);
249  }
250  if (streq("help", options[idx].name)) {
251  usage(0);
252  }
253  if (streq("active-low", options[idx].name) ||
254  streq("low", options[idx].name)) {
255  }
256  else if (streq("bias", options[idx].name)) {
257  base_flags |= get_bias(optarg);
258  }
259  else if (streq("name", options[idx].name)) {
260  consumer_name = optarg;
261  }
262  else if (streq("output-drive", options[idx].name)) {
263  base_flags |= get_output_drive(optarg);
264  }
265  else if (streq("wait", options[idx].name)) {
266  wait = true;
267  if (!read_int(optarg, &seconds)) {
268  fprintf(stderr, "%s: invalid value for seconds: %s\n\n",
269  THIS_EXECUTABLE, optarg);
270  usage(EINVAL);
271  }
272  }
273  else {
274  fprintf(stderr, "%s: unhandled option: %s\n\n",
275  THIS_EXECUTABLE, options[idx].name);
276  usage(EINVAL);
277  }
278  break;
279  case 'b':
280  base_flags |= get_bias(optarg);
281  continue;
282  case 'h':
283  usage(0);
284  case 'l':
285  active_low = true;
286  continue;
287  case 'n':
288  consumer_name = optarg;
289  continue;
290  case 'o':
291  drive = get_output_drive(optarg);
292  printf("Drive: 0x%" PRIx64 ", base: 0x:%" PRIx64 "\n",
293  drive, base_flags);
294 
295  base_flags |= drive;
296  continue;
297  case 'v':
298  version("THIS_EXECUTABLE");
299  case 'w':
300  wait = true;
301  if (!read_int(optarg, &seconds)) {
302  fprintf(stderr, "%s: invalid value for seconds: %s\n\n",
303  THIS_EXECUTABLE, optarg);
304  usage(EINVAL);
305  }
306  continue;
307  default:
308  usage(EINVAL);
309  }
310  }
311  if (optind >= argc) {
312  fprintf(stderr, "%s: No gpio chip id provided.\n", THIS_EXECUTABLE);
313  usage(EINVAL);
314  }
315 
316  if (active_low) {
317  base_flags |= GPIO_V2_LINE_FLAG_ACTIVE_LOW;
318  }
319 
320  request = get_gpio_request(argv[optind], consumer_name,
321  GPIO_V2_LINE_FLAG_OUTPUT);
322 
323  /* Now handle each line argument in turn. */
324  if (((argc - 1) - optind) > GPIO_V2_LINES_MAX) {
325  fprintf(stderr, "%s: maximum gpio lines (%d) exceeded.\n",
326  THIS_EXECUTABLE, GPIO_V2_LINES_MAX);
327  exit(EINVAL);
328  }
329 
330  for (idx = optind + 1; idx < argc; idx++) {
331  line_flags = base_flags;
332 
333  if (!read_line_spec(argv[idx], &line, &line_value, &line_flags,
334  LINE_FLAG_BIAS_MASK |
335  LINE_FLAG_OUTPUT_DRIVER_MASK |
336  LINE_FLAG_ACTIVE_LOW_MASK)) {
337  fprintf(stderr, "expecting numeric gpio line with value and"
338  "optional flags: \"%s\"\n", argv[idx]);
339  usage(EINVAL);
340  }
341  line_name = bgpio_configure_line(request, line,
342  line_flags, line_value);
343  if (!line_name) {
344  fprintf(stderr, "%s: unable to get line (%d) for chip\n",
345  THIS_EXECUTABLE, line);
346  exit(EINVAL);
347  }
348  }
349 
350  if (request->req.num_lines) {
351  err = bgpio_complete_request(request);
352  if (err) {
353  fprintf(stderr, "%s: error completing bgpio_request: %s\n",
354  THIS_EXECUTABLE, strerror(err));
355  exit(err);
356  }
357  if (wait) {
358  do {
359  /* If seconds is zero, we want to wait forever.
360  * Waking once an hour (3600 seconds) should not be
361  * too much overhead. */
362  sleep(seconds? seconds: 3600);
363  } while (seconds == 0);
364  }
365  }
366 
367  exit(bgpio_close_request(request));
368 }
